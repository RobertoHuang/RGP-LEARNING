# 枚举类模板
```
public enum DictionaryTypeEnum {
    APP_FACTORY("DD00001","应用工厂构建");

    private final String code;
    private final String value;

    private DictionaryTypeEnum(String code, String label) {
        this.code = code;
        this.value = label;

    }

    public String getCode() {
        return code;
    }

    public String getValue() {
        return value;
    }

    public DictionaryTypeEnum returnEnum(String code) {
        return getByCode(code);
    }

    public static DictionaryTypeEnum getByCode(String code) {
        if (code == null) {
            return null;
        }

        if (values() == null) {
            return null;
        }

        for (DictionaryTypeEnum t : values()) {
            if (t.getCode().equals(code)) {
                return t;
            }
        }
        return null;
    }
}
```
